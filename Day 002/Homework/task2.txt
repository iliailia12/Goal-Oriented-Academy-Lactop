# String (სტრინგი) არის ტექსტური მონაცემი, რომელიც მოთავსებულია ბრჭყალებში.
# იგი გამოიყენება ასოების, სიტყვების ან სიმბოლოების დასამოსად.
text = "Hello, World!"   # ეს არის string-ის მაგალითი


# Integer (ინტეჯერი) არის მთელი რიცხვი, რომელშიც არ არის ათწილადი.
# გამოიყენება რაოდენობების, მთელ რიცხვობრივ მნიშვნელობების გამოსახატად.
number_int = 25          # ეს არის integer-ის მაგალითი


# Float (ფლოათი) არის ათწილადი რიცხვი.
# გამოიყენება მაშინ, როცა საჭიროა რიცხვი წერტილით (ათწილადით).
number_float = 3.14      # ეს არის float-ის მაგალითი
